include<stdio.h>
int product=0,i=0;
int multiply(int,int);
int main(){
    int a,b,product;
    printf("enter any two integers");
    scanf("%d %d",&a,&b);
    product=multiply(a,b);
    printf("multiplication of two integers is %d",product);
}
int multiply(int a,int b)
{
    if(i<a)
    {
        product=product+b;
        i++;
        multiply(a,b);
    }
    return product;
}
